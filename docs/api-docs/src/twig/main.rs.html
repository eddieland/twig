28fn main() -> Result<()> {
29  // Set up the no-worries panic handler with custom configuration
30  let config: NoWorriesConfig = NoWorriesConfig {
31    metadata: NoWorriesMetadata {
32      name: "twig".to_string(),
33      support_email: Some("ejones@lat.ai".to_string()),
34      // Other metadata fields use defaults from Cargo.toml
35      ..Default::default()
36    },
37    ..Default::default()
38  };
39  no_worries!(config).expect("Failed to set up panic handler");
40
41  // Parse CLI arguments using the derive-based implementation
42  let cmd = cli::Cli::parse();
43
44  // Set up tracing based on verbosity level
45  let verbose_count = cmd.verbose;
46  let level = match verbose_count {
47    0 => tracing::Level::WARN,  // Default: warnings and errors
48    1 => tracing::Level::INFO,  // -v: info, warnings, and errors
49    2 => tracing::Level::DEBUG, // -vv: debug, info, warnings, and errors
50    _ => tracing::Level::TRACE, // -vvv or more: trace and everything else
51  };
52
53  // Initialize the tracing subscriber with the specified level
54  tracing_subscriber::registry()
55    .with(fmt::layer())
56    .with(EnvFilter::from_default_env().add_directive(level.into()))
57    .init();
58
59  debug!("Tracing initialized with level: {}", level);
60
61  handle_cli(cmd)
62}
