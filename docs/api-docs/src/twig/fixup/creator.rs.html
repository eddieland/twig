1//! # Fixup Commit Creation
2//!
3//! This module handles the creation of fixup commits that target existing
4//! commits in the Git history. It provides functionality to create properly
5//! formatted fixup commits that can be automatically squashed during
6//! interactive rebase.
7
8use std::path::Path;
9use std::process::Command;
10
11use anyhow::{Context, Result};
12use git2::Repository;
13use twig_core::output::{print_error, print_warning};
14
15use crate::consts;
16use crate::fixup::commit_collector::CommitCandidate;

18/// Create a fixup commit for the selected target commit
19pub fn create_fixup_commit(repo_path: &Path, target_commit: &CommitCandidate) -> Result<()> {
20  // Check if there are staged changes
21  if !has_staged_changes(repo_path)? {
22    print_warning("No staged changes found. Stage changes first before creating a fixup commit.");
23    return Ok(());
24  }
25
26  tracing::debug!("Creating fixup commit for {}", target_commit.hash);
27
28  // Create the fixup commit
29  let output = Command::new(consts::GIT_EXECUTABLE)
30    .args(["commit", "--fixup", &target_commit.hash])
31    .current_dir(repo_path)
32    .output()
33    .context("Failed to execute git commit --fixup command")?;
34
35  if output.status.success() {
36    tracing::debug!("Fixup commit created successfully");
37    tracing::trace!("Git output: {}", String::from_utf8_lossy(&output.stdout));
38    Ok(())
39  } else {
40    let stderr = String::from_utf8_lossy(&output.stderr);
41    print_error("Failed to create fixup commit.");
42    tracing::warn!("Git commit --fixup failed: {}", stderr);
43    Err(anyhow::anyhow!("Git commit --fixup command failed: {}", stderr))
44  }
45}

47/// Checks if there are staged changes in the Git repository.
48///
49/// This function uses git2 to determine if there are any staged changes
50/// ready to be committed. This is essential before creating a fixup commit,
51/// as Git requires changes to commit.
52///
53/// # Arguments
54///
55/// * `repo_path` - Path to the Git repository root
56///
57/// # Returns
58///
59/// Returns `true` if there are staged changes, `false` if the staging area is
60/// clean.
61///
62/// # Errors
63///
64/// Returns an error if the repository path is invalid or git2 operations fail.
65pub fn has_staged_changes(repo_path: &Path) -> Result<bool> {
66  let repo = Repository::open(repo_path).context("Failed to open git repository")?;
67
68  // Get the index (staging area)
69  let index = repo.index().context("Failed to get repository index")?;
70
71  // Get HEAD tree to compare against
72  let head = repo.head().context("Failed to get HEAD reference")?;
73  let head_commit = head.peel_to_commit().context("Failed to get HEAD commit")?;
74  let head_tree = head_commit.tree().context("Failed to get HEAD tree")?;
75
76  // Compare the index with HEAD tree to see if there are staged changes
77  let diff = repo
78    .diff_tree_to_index(Some(&head_tree), Some(&index), None)
79    .context("Failed to create diff between HEAD and index")?;
80
81  // If there are any deltas (changes), then there are staged changes
82  Ok(diff.deltas().len() > 0)
83}
