1//! # Configuration Management
2//!
3//! Handles application configuration, directory management, and settings
4//! for the twig tool, including XDG base directory support.
5
6use std::fs;
7use std::path::{Path, PathBuf};
8
9use anyhow::{Context, Result};
10use directories::ProjectDirs;
11
12/// Represents the configuration directories for the twig application
13pub struct ConfigDirs {
14  pub config_dir: PathBuf,
15  pub data_dir: PathBuf,
16  pub cache_dir: Option<PathBuf>,
17}
18
19impl ConfigDirs {
20  /// Create a new ConfigDirs instance
21  pub fn new() -> Result<Self> {
22    let proj_dirs = ProjectDirs::from("eddieland", "", "twig").context("Failed to determine project directories")?;
23
24    let config_dir = proj_dirs.config_dir().to_path_buf();
25    let data_dir = proj_dirs.data_dir().to_path_buf();
26    let cache_dir = Some(proj_dirs.cache_dir().to_path_buf());
27
28    Ok(Self {
29      config_dir,
30      data_dir,
31      cache_dir,
32    })
33  }
34
35  /// Get the config directory
36  pub fn config_dir(&self) -> &PathBuf {
37    &self.config_dir
38  }
39
40  /// Get the data directory
41  pub fn data_dir(&self) -> &PathBuf {
42    &self.data_dir
43  }
44
45  /// Get the cache directory
46  pub fn cache_dir(&self) -> Option<&PathBuf> {
47    self.cache_dir.as_ref()
48  }
49
50  /// Initialize the configuration directories
51  pub fn init(&self) -> Result<()> {
52    fs::create_dir_all(&self.config_dir).context("Failed to create config directory")?;
53    fs::create_dir_all(&self.data_dir).context("Failed to create data directory")?;
54
55    // Create an empty registry file if it doesn't exist
56    let registry_path = self.registry_path();
57    if !registry_path.exists() {
58      fs::write(&registry_path, "[]").context("Failed to create empty registry file")?;
59    }
60
61    Ok(())
62  }
63
64  /// Get the path to the registry file
65  pub fn registry_path(&self) -> PathBuf {
66    self.data_dir.join("registry.json")
67  }
68
69  /// Get the path to the repository-local state directory
70  pub fn repo_state_dir<P: AsRef<Path>>(&self, repo_path: P) -> PathBuf {
71    repo_path.as_ref().join(".twig")
72  }
73
74  /// Get the path to the repository-local state file
75  pub fn repo_state_path<P: AsRef<Path>>(&self, repo_path: P) -> PathBuf {
76    self.repo_state_dir(repo_path).join("state.json")
77  }
78}
79
80/// Initialize the configuration directories
81pub fn init() -> Result<()> {
82  use crate::utils::output::{format_repo_path, print_success};
83
84  let config_dirs = ConfigDirs::new()?;
85  config_dirs.init()?;
86
87  print_success("Initialized twig configuration directories:");
88  println!(
89    "  Config: {}",
90    format_repo_path(&config_dirs.config_dir.display().to_string())
91  );
92  println!(
93    "  Data: {}",
94    format_repo_path(&config_dirs.data_dir.display().to_string())
95  );
96
97  Ok(())
98}

100/// Get the configuration directories
101pub fn get_config_dirs() -> Result<ConfigDirs> {
102  ConfigDirs::new()
103}

112  #[test]

114    // Use TestEnv from twig-test-utils to properly manage environment variables
115    let _test_env = EnvTestGuard::new();
116
117    let config_dirs = ConfigDirs::new().unwrap();
118    let _ = config_dirs.init();
119
120    assert!(config_dirs.config_dir().exists() || config_dirs.config_dir().parent().unwrap().exists());
121    assert!(config_dirs.data_dir().exists() || config_dirs.data_dir().parent().unwrap().exists());
122    assert!(config_dirs.cache_dir().is_some());
123  }
