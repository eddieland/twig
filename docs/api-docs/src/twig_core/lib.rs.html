1//! # Twig Core Library
2//!
3//! Core library for twig plugins providing configuration structures, state
4//! models, and utility functions for plugin developers. This crate enables
5//! plugins to access twig's configuration and state in a read-only manner while
6//! maintaining their own separate state.
7
8pub mod config;
9pub mod git;
10pub mod output;
11pub mod state;
12pub mod tree_renderer;
13pub mod utils;
14
15// Re-export main types for plugin developers
16pub use config::{ConfigDirs, get_config_dirs};
17pub use git::{current_branch, detect_repository, detect_repository_from_path, in_git_repository};
18pub use output::{ColorMode, format_repo_path, print_error, print_info, print_success, print_warning};
19pub use state::{BranchDependency, BranchMetadata, Registry, RepoState, Repository, RootBranch, create_worktree};
20pub use utils::{get_current_branch_github_pr, get_current_branch_jira_issue};
21
22/// Plugin-specific utilities
23pub mod plugin {
24  use std::path::PathBuf;
25
26  use anyhow::Result;
27
28  pub use super::config::get_config_dirs;
29  pub use super::git::current_branch;
30
31  /// Get plugin-specific config directory
32  pub fn plugin_config_dir(plugin_name: &str) -> Result<PathBuf> {
33    let config_dirs = get_config_dirs()?;
34    Ok(config_dirs.config_dir().join("plugins").join(plugin_name))
35  }
36
37  /// Get plugin-specific data directory
38  pub fn plugin_data_dir(plugin_name: &str) -> Result<PathBuf> {
39    let config_dirs = get_config_dirs()?;
40    Ok(config_dirs.data_dir().join("plugins").join(plugin_name))
41  }
42}
