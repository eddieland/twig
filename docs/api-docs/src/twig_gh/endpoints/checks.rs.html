1use anyhow::{Context, Result};
2use reqwest::{StatusCode, header};
3use serde::Deserialize;
4use tracing::instrument;
5
6use crate::client::GitHubClient;
7use crate::consts::{ACCEPT, USER_AGENT};
8use crate::models::CheckRun;
9
10impl GitHubClient {
11  /// Get check runs for a pull request
12  #[instrument(skip(self), level = "debug")]
13  pub async fn get_check_runs(&self, owner: &str, repo: &str, ref_sha: &str) -> Result<Vec<CheckRun>> {
14    let url = format!(
15      "{}/repos/{}/{}/commits/{}/check-runs",
16      self.base_url, owner, repo, ref_sha
17    );
18
19    let response = self
20      .client
21      .get(&url)
22      .header(header::ACCEPT, ACCEPT)
23      .header(header::USER_AGENT, USER_AGENT)
24      .basic_auth(&self.auth.username, Some(&self.auth.token))
25      .send()
26      .await
27      .context("Failed to fetch check runs")?;
28
29    #[derive(Deserialize)]
30    struct CheckRunsResponse {
31      check_runs: Vec<CheckRun>,
32    }
33
34    match response.status() {
35      StatusCode::OK => {
36        // First get the response body as text
37        let body = response.text().await.context("Failed to read response body")?;
38
39        // Then try to parse it as JSON
40        let check_runs_response = match serde_json::from_str::<CheckRunsResponse>(&body) {
41          Ok(response) => response,
42          Err(e) => {
43            // Try to extract the error message from the response
44            if let Ok(error_json) = serde_json::from_str::<serde_json::Value>(&body) {
45              if let Some(message) = error_json.get("message").and_then(|m| m.as_str()) {
46                return Err(anyhow::anyhow!(
47                  "Failed to parse check runs: GitHub API error: {}",
48                  message
49                ));
50              }
51            }
52            // Fall back to the original error if we can't extract a message
53            return Err(anyhow::anyhow!("Failed to parse check runs: {}", e));
54          }
55        };
56
57        Ok(check_runs_response.check_runs)
58      }
59      StatusCode::NOT_FOUND => Err(anyhow::anyhow!("Commit {} not found", ref_sha)),
60      StatusCode::UNAUTHORIZED | StatusCode::FORBIDDEN => Err(anyhow::anyhow!(
61        "Authentication failed. Please check your GitHub credentials."
62      )),
63      _ => Err(anyhow::anyhow!(
64        "Unexpected error: HTTP {} - {}",
65        response.status(),
66        response.text().await.unwrap_or_default()
67      )),
68    }
69  }
70}

79  #[tokio::test]
80  async fn test_get_check_runs() -> anyhow::Result<()> {
81    let mock_server = MockServer::start().await;
82    let auth = GitHubAuth {
83      username: "test_user".to_string(),
84      token: "test_token".to_string(),
85    };
86    let mut client = GitHubClient::new(auth);
87    client.base_url = mock_server.uri();
88
89    let ref_sha = "abc123def456";
90
91    // Mock response for check runs
92    Mock::given(method("GET"))
93      .and(path(format!("/repos/owner/repo/commits/{}/check-runs", ref_sha)))
94      .and(header(header::ACCEPT, ACCEPT))
95      .and(header(header::USER_AGENT, USER_AGENT))
96      .and(header(header::AUTHORIZATION, "Basic dGVzdF91c2VyOnRlc3RfdG9rZW4="))
97      .respond_with(ResponseTemplate::new(200).set_body_json(serde_json::json!({
98          "total_count": 2,
99          "check_runs": [
100              {
101                  "id": 1,
102                  "name": "test-suite",
103                  "status": "completed",
104                  "conclusion": "success",
105                  "started_at": "2023-01-01T00:00:00Z",
106                  "completed_at": "2023-01-01T00:01:00Z"
107              },
108              {
109                  "id": 2,
110                  "name": "lint",
111                  "status": "completed",
112                  "conclusion": "failure",
113                  "started_at": "2023-01-01T00:00:00Z",
114                  "completed_at": "2023-01-01T00:01:00Z"
115              }
116          ]
117      })))
118      .mount(&mock_server)
119      .await;
120
121    let check_runs = client.get_check_runs("owner", "repo", ref_sha).await?;
122    assert_eq!(check_runs.len(), 2);
123
124    // Verify first check run
125    assert_eq!(check_runs[0].id, 1);
126    assert_eq!(check_runs[0].name, "test-suite");
127    assert_eq!(check_runs[0].status, "completed");
128    assert_eq!(check_runs[0].conclusion, Some("success".to_string()));
129
130    // Verify second check run
131    assert_eq!(check_runs[1].id, 2);
132    assert_eq!(check_runs[1].name, "lint");
133    assert_eq!(check_runs[1].status, "completed");
134    assert_eq!(check_runs[1].conclusion, Some("failure".to_string()));
135
136    Ok(())
137  }
