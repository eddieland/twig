1//! HOME directory management for testing
2//!
3//! This module provides utilities for isolating HOME directory during testing
4//! to prevent tests from interfering with the user's actual home directory.
5
6use std::env;
7use std::path::{Path, PathBuf};
8
9use tempfile::TempDir;
10
11/// A test environment that overrides the HOME directory to use a temporary
12/// directory This is useful for testing credential management and other
13/// home directory dependent functionality
14pub struct HomeEnvTestGuard {
15  /// The temporary directory that will be used as HOME
16  pub temp_dir: TempDir,
17  /// The original HOME value
18  original_home: String,
19}
20
21impl Default for HomeEnvTestGuard {
22  fn default() -> Self {
23    Self::new()
24  }
25}
26
27impl HomeEnvTestGuard {
28  /// Create a new test environment with a temporary HOME directory
29  pub fn new() -> Self {
30    let temp_dir = TempDir::new().expect("Failed to create temporary directory");
31
32    // Save original HOME environment variable
33    let original_home = env::var("HOME").expect("HOME environment variable must be set");
34
35    // Override HOME to use the temporary directory
36    unsafe {
37      env::set_var("HOME", temp_dir.path());
38    }
39
40    Self {
41      temp_dir,
42      original_home,
43    }
44  }
45
46  /// Get the path to the temporary HOME directory
47  pub fn home_dir(&self) -> &Path {
48    self.temp_dir.path()
49  }
50
51  /// Get the path to a file in the temporary HOME directory
52  pub fn home_path(&self, relative_path: &str) -> PathBuf {
53    self.temp_dir.path().join(relative_path)
54  }
55}
56
57impl Drop for HomeEnvTestGuard {
58  fn drop(&mut self) {
59    // Restore original HOME environment variable
60    unsafe {
61      env::set_var("HOME", &self.original_home);
62    }
63  }
64}
