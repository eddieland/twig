11/// Execute a plugin with the given name and arguments
12pub fn execute_plugin(plugin_name: &str, args: Vec<String>, verbosity: u8) -> Result<()> {
13  let plugin_binary = format!("twig-{plugin_name}");
14
15  // Check if plugin exists in PATH
16  if !plugin_exists(&plugin_binary)? {
17    return Err(anyhow::anyhow!(
18      "Unknown command '{}'. No plugin 'twig-{}' found in PATH.\n\n\
19             To install plugins, place executable files named 'twig-<command>' in your PATH.",
20      plugin_name,
21      plugin_name
22    ));
23  }
24
25  // Set up environment variables
26  let config_dirs = twig_core::get_config_dirs()?;
27  let current_repo = twig_core::detect_repository();
28  let current_branch = twig_core::current_branch().unwrap_or(None);
29
30  let mut cmd = Command::new(&plugin_binary);
31  cmd
32    .args(args)
33    .env("TWIG_CONFIG_DIR", config_dirs.config_dir())
34    .env("TWIG_DATA_DIR", config_dirs.data_dir())
35    .env("TWIG_VERSION", env!("CARGO_PKG_VERSION"))
36    .env("TWIG_VERBOSITY", verbosity.to_string())
37    .stdin(Stdio::inherit())
38    .stdout(Stdio::inherit())
39    .stderr(Stdio::inherit());
40
41  if let Some(repo) = current_repo {
42    cmd.env("TWIG_CURRENT_REPO", repo.display().to_string());
43  }
44
45  if let Some(branch) = current_branch {
46    cmd.env("TWIG_CURRENT_BRANCH", branch);
47  }
48
49  let status = cmd
50    .status()
51    .with_context(|| format!("Failed to execute plugin '{plugin_binary}'"))?;
52
53  std::process::exit(status.code().unwrap_or(1));
54}

56/// Check if a plugin exists in PATH
57fn plugin_exists(plugin_name: &str) -> Result<bool> {
58  // Try 'which' command (Unix-like systems)
59  if let Ok(output) = Command::new("which").arg(plugin_name).output() {
60    return Ok(output.status.success());
61  }
62
63  // Try 'where' command (Windows)
64  if let Ok(output) = Command::new("where").arg(plugin_name).output() {
65    return Ok(output.status.success());
66  }
67
68  // Fallback: try to execute the command with --help
69  if let Ok(output) = Command::new(plugin_name).arg("--help").output() {
70    return Ok(output.status.success());
71  }
72
73  Ok(false)
74}

76/// List available plugins in PATH
77#[allow(dead_code)]
78pub fn list_available_plugins() -> Result<Vec<String>> {
79  let mut plugins = Vec::new();
80
81  // Get PATH environment variable
82  let path_var = env::var("PATH").unwrap_or_default();
83  let paths: Vec<&str> = if cfg!(windows) {
84    path_var.split(';').collect()
85  } else {
86    path_var.split(':').collect()
87  };
88
89  for path_str in paths {
90    let path = std::path::Path::new(path_str);
91    if !path.exists() {
92      continue;
93    }
94
95    if let Ok(entries) = std::fs::read_dir(path) {
96      for entry in entries.flatten() {
97        let file_name = entry.file_name();
98        let file_name_str = file_name.to_string_lossy();
99
100        // Check if it's a twig plugin
101        if file_name_str.starts_with("twig-") {
102          let plugin_name = file_name_str.strip_prefix("twig-").unwrap();
103
104          // Remove file extension on Windows
105          let plugin_name = if cfg!(windows) && plugin_name.ends_with(".exe") {
106            plugin_name.strip_suffix(".exe").unwrap()
107          } else {
108            plugin_name
109          };
110
111          if !plugins.contains(&plugin_name.to_string()) {
112            plugins.push(plugin_name.to_string());
113          }
114        }
115      }
116    }
117  }
118
119  plugins.sort();
120  Ok(plugins)
121}

123/// Generate suggestions for unknown commands
124#[allow(dead_code)]
125pub fn suggest_similar_commands(unknown_command: &str, available_plugins: &[String]) -> Vec<String> {
126  let mut suggestions = Vec::new();
127
128  // Built-in commands that might be similar
129  let builtin_commands = [
130    "branch",
131    "cascade",
132    "commit",
133    "completion",
134    "creds",
135    "dashboard",
136    "git",
137    "github",
138    "jira",
139    "rebase",
140    "switch",
141    "sync",
142    "tree",
143    "worktree",
144  ];
145
146  // Combine built-in commands and plugins
147  let mut all_commands: Vec<&str> = builtin_commands.to_vec();
148  for plugin in available_plugins {
149    all_commands.push(plugin.as_str());
150  }
151
152  // Simple string distance matching
153  for command in all_commands.iter() {
154    if levenshtein_distance(unknown_command, command) <= 2 {
155      suggestions.push(command.to_string());
156    }
157  }
158
159  // If no close matches, suggest commands that start with the same letter
160  if suggestions.is_empty() {
161    let first_char = unknown_command
162      .chars()
163      .next()
164      .unwrap_or('\0')
165      .to_lowercase()
166      .next()
167      .unwrap_or('\0');
168    for command in all_commands.iter() {
169      if command
170        .chars()
171        .next()
172        .unwrap_or('\0')
173        .to_lowercase()
174        .next()
175        .unwrap_or('\0')
176        == first_char
177      {
178        suggestions.push(command.to_string());
179      }
180    }
181  }
182
183  suggestions.sort();
184  suggestions.dedup();
185  suggestions.truncate(5); // Limit to 5 suggestions
186  suggestions
187}

189/// Calculate Levenshtein distance between two strings
190#[allow(dead_code)]
191#[allow(clippy::needless_range_loop)] // iterator approach hurts readability
192fn levenshtein_distance(s1: &str, s2: &str) -> usize {
193  let len1 = s1.len();
194  let len2 = s2.len();
195
196  if len1 == 0 {
197    return len2;
198  }
199  if len2 == 0 {
200    return len1;
201  }
202
203  let mut matrix = vec![vec![0; len2 + 1]; len1 + 1];
204
205  for i in 0..=len1 {
206    matrix[i][0] = i;
207  }
208  for j in 0..=len2 {
209    matrix[0][j] = j;
210  }
211
212  let s1_chars: Vec<char> = s1.chars().collect();
213  let s2_chars: Vec<char> = s2.chars().collect();
214
215  for i in 1..=len1 {
216    for j in 1..=len2 {
217      let cost = if s1_chars[i - 1] == s2_chars[j - 1] { 0 } else { 1 };
218      matrix[i][j] = std::cmp::min(
219        std::cmp::min(
220          matrix[i - 1][j] + 1, // deletion
221          matrix[i][j - 1] + 1, // insertion
222        ),
223        matrix[i - 1][j - 1] + cost, // substitution
224      );
225    }
226  }
227
228  matrix[len1][len2]
229}

231#[cfg(test)]
232mod tests {
233  use super::*;
234
235  #[test]
236  fn test_levenshtein_distance() {
237    assert_eq!(levenshtein_distance("", ""), 0);
238    assert_eq!(levenshtein_distance("abc", "abc"), 0);
239    assert_eq!(levenshtein_distance("abc", "ab"), 1);
240    assert_eq!(levenshtein_distance("abc", "def"), 3);
241    assert_eq!(levenshtein_distance("branch", "brach"), 1);
242    assert_eq!(levenshtein_distance("deploy", "deploi"), 1);
243  }
244
245  #[test]
246  fn test_suggest_similar_commands() {
247    let plugins = vec!["deploy".to_string(), "backup".to_string()];
248
249    let suggestions = suggest_similar_commands("deploi", &plugins);
250    assert!(suggestions.contains(&"deploy".to_string()));
251
252    let suggestions = suggest_similar_commands("brach", &plugins);
253    assert!(suggestions.contains(&"branch".to_string()));
254  }
