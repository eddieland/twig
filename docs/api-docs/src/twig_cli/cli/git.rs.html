1//! # Git Command
2//!
3//! Derive-based implementation of the git command for managing Git
4//! repositories.
5
6use anyhow::{Result, anyhow};
7use clap::{Args, Subcommand};
8
9/// Command for Git repository management
10#[derive(Args)]
11pub struct GitArgs {
12  /// The subcommand to execute
13  #[command(subcommand)]
14  pub subcommand: GitSubcommands,
15}
16
17/// Subcommands for the git command
18#[derive(Subcommand)]
19pub enum GitSubcommands {
20  /// Add a repository to the registry
21  #[command(long_about = "Registers a Git repository with twig for tracking and management.\n\n\
22                     Once added, the repository can be referenced in other twig commands and\n\
23                     included in batch operations. The repository must be a valid Git repository\n\
24                     with proper credentials configured if needed for remote operations.")]
25  Add(AddCommand),
26
27  /// Execute a git command in repositories
28  #[command(long_about = "Executes a Git command in one or all registered repositories.\n\n\
29                     This powerful feature allows you to run the same Git operation across\n\
30                     multiple repositories simultaneously. The command is executed as-is,\n\
31                     so ensure it's a valid Git command. Credentials may be required\n\
32                     depending on the Git operation being performed.")]
33  Exec(ExecCommand),
34
35  /// Fetch updates for repositories
36  #[command(long_about = "Fetches updates from remote repositories.\n\n\
37                     This updates local references to remote branches without modifying your\n\
38                     working directory. Requires proper Git credentials to be configured for\n\
39                     repositories with private remotes.")]
40  Fetch(FetchCommand),
41
42  /// List all repositories in the registry
43  #[command(long_about = "Displays all Git repositories currently registered with twig.\n\n\
44                     Shows the repository paths and any additional tracking information\n\
45                     to help you manage your repositories.")]
46  #[command(alias = "ls")]
47  List,
48
49  /// Remove a repository from the registry
50  #[command(
51    long_about = "Removes a previously registered Git repository from twig's tracking.\n\n\
52                     This only affects twig's registry and does not delete or modify the\n\
53                     actual repository files."
54  )]
55  #[command(alias = "rm")]
56  Remove(RemoveCommand),
57
58  /// List stale branches in repositories
59  #[command(long_about = "Identifies and lists branches that haven't been updated recently.\n\n\
60                     This helps you identify abandoned or forgotten branches that might need\n\
61                     attention, cleanup, or merging. The command analyzes local branch information\n\
62                     and doesn't require remote credentials unless combined with a fetch operation.")]
63  #[command(alias = "stale")]
64  StaleBranches(StaleBranchesCommand),
65}

127/// Handle the git command
128///
129/// This function processes the git subcommands and executes the
130/// corresponding actions such as adding, removing, listing repositories,
131/// fetching updates, executing commands, and finding stale branches.
132pub(crate) fn handle_git_command(git: GitArgs) -> Result<()> {
133  match git.subcommand {
134    GitSubcommands::Add(cmd) => crate::git::add_repository(&cmd.path),
135    GitSubcommands::Exec(cmd) => {
136      if cmd.all {
137        crate::git::execute_all_repositories(&cmd.command)
138      } else {
139        let repo_arg = cmd.repo.as_deref();
140        let repo_path = crate::utils::resolve_repository_path(repo_arg)?;
141        crate::git::execute_repository(repo_path, &cmd.command)
142      }
143    }
144    GitSubcommands::Fetch(cmd) => {
145      if cmd.all {
146        crate::git::fetch_all_repositories()
147      } else {
148        let repo_arg = cmd.repo.as_deref();
149        let repo_path = crate::utils::resolve_repository_path(repo_arg)?;
150        crate::git::fetch_repository(repo_path, true)
151      }
152    }
153    GitSubcommands::List => crate::git::list_repositories(),
154    GitSubcommands::Remove(cmd) => crate::git::remove_repository(&cmd.path),
155    GitSubcommands::StaleBranches(cmd) => {
156      let days = cmd
157        .days
158        .parse::<u32>()
159        .map_err(|e| anyhow!("Days must be a positive number: {}", e))?;
160
161      if cmd.all {
162        crate::git::find_stale_branches_all(days)
163      } else {
164        let repo_arg = cmd.repo.as_deref();
165        let repo_path = crate::utils::resolve_repository_path(repo_arg)?;
166        crate::git::find_stale_branches(repo_path, days)
167      }
168    }
169  }
