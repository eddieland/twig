1//! # Client Creation
2//!
3//! Centralized client creation for external services like GitHub and Jira.
4//! This module provides helper functions to create authenticated clients
5//! with proper error handling and credential management.
6
7// Re-export the original client creation functions
8use anyhow::{Context, Result};
9use tokio::runtime::Runtime;
10use twig_gh::GitHubClient;
11pub use twig_gh::create_github_client;
12use twig_jira::JiraClient;
13pub use twig_jira::create_jira_client;
14
15use crate::consts::{DEFAULT_JIRA_HOST, ENV_JIRA_HOST};
16use crate::creds::{get_github_credentials, get_jira_credentials};
17
18/// Creates an authenticated GitHub client using credentials from .netrc
19///
20/// This function handles retrieving GitHub credentials and creating a client
21/// in one step, with proper error handling.
22pub fn create_github_client_from_netrc() -> Result<GitHubClient> {
23  let credentials = get_github_credentials().context("Failed to get GitHub credentials")?;
24
25  Ok(create_github_client(&credentials.username, &credentials.password))
26}
27
28/// Creates an authenticated Jira client using credentials from .netrc
29///
30/// This function handles retrieving Jira credentials, determining the
31/// Jira host URL, and creating a client in one step, with proper error
32/// handling.
33pub fn create_jira_client_from_netrc() -> Result<JiraClient> {
34  let credentials = get_jira_credentials().context("Failed to get Jira credentials")?;
35
36  // Get Jira host from environment or use default
37  let jira_host = std::env::var(ENV_JIRA_HOST).unwrap_or_else(|_| DEFAULT_JIRA_HOST.to_string());
38
39  Ok(create_jira_client(
40    &jira_host,
41    &credentials.username,
42    &credentials.password,
43  ))
44}
45
46/// Creates a tokio runtime and an authenticated GitHub client
47///
48/// This is a convenience function for CLI commands that need both a runtime
49/// and a GitHub client.
50pub fn create_github_runtime_and_client() -> Result<(Runtime, GitHubClient)> {
51  let rt = Runtime::new().context("Failed to create async runtime")?;
52  let client = create_github_client_from_netrc()?;
53  Ok((rt, client))
54}
55
56/// Creates a tokio runtime and an authenticated Jira client
57///
58/// This is a convenience function for CLI commands that need both a runtime
59/// and a Jira client.
60pub fn create_jira_runtime_and_client() -> Result<(Runtime, JiraClient)> {
61  let rt = Runtime::new().context("Failed to create async runtime")?;
62  let client = create_jira_client_from_netrc()?;
63  Ok((rt, client))
64}
