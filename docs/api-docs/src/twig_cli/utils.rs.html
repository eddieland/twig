6use std::path::PathBuf;
7
8use anyhow::{Context, Result};
9use twig_core::{detect_repository, detect_repository_from_path};
10
11/// Resolve a repository path from a command line argument or current directory
12pub fn resolve_repository_path(repo_arg: Option<&str>) -> Result<PathBuf> {
13  match repo_arg {
14    Some(path) => {
15      let path_buf = PathBuf::from(path);
16      if !path_buf.exists() {
17        return Err(anyhow::anyhow!("Repository path does not exist: {}", path));
18      }
19      detect_repository_from_path(&path_buf).context(format!("Failed to detect repository at path: {path}"))
20    }
21    None => {
22      // Try to detect the current repository
23      detect_repository().context("No repository specified and not in a git repository")
24    }
25  }
26}

28#[cfg(test)]
29mod tests {
30  use tempfile::TempDir;
31  use twig_test_utils::GitRepoTestGuard;
32
33  use super::*;
34
35  // Test resolve_repository_path with a valid path
36  #[test]
37  fn test_resolve_repository_path_with_valid_path() {
38    // Create a temporary directory to use as our "repository"
39    let temp_dir = TempDir::new().unwrap();
40
41    // This is a bit of a hack, but we can't easily mock these functions
42    // without changing the code structure, so we'll just test the error path
43    let result = resolve_repository_path(Some(temp_dir.path().to_str().unwrap()));
44
45    // If the path exists but isn't a git repo, we'll get an error about failing to
46    // detect repository
47    assert!(result.is_err());
48    assert!(result.unwrap_err().to_string().contains("Failed to detect repository"));
49  }
50
51  // Test resolve_repository_path with an invalid path
52  #[test]
53  fn test_resolve_repository_path_with_invalid_path() {
54    let result = resolve_repository_path(Some("/path/that/does/not/exist"));
55    assert!(result.is_err());
56    assert!(
57      result
58        .unwrap_err()
59        .to_string()
60        .contains("Repository path does not exist")
61    );
62  }
63
64  // Test resolve_repository_path with None (current directory)
65  #[test]
66  fn test_resolve_repository_path_with_none() {
67    // Create a temporary git repository and change to its directory
68    let git_repo = GitRepoTestGuard::new_and_change_dir();
69    let git_repo_path = std::fs::canonicalize(git_repo.path()).unwrap();
70
71    // Now test the function with None
72    let result = resolve_repository_path(None);
73
74    // The result should be Ok and contain our temporary directory path
75    assert!(result.is_ok());
76    let repo_path = std::fs::canonicalize(result.unwrap()).unwrap();
77    assert_eq!(repo_path, git_repo_path);
78  }
