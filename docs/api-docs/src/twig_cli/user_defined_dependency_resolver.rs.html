1//! # User-Defined Dependency Resolver
2//!
3//! Resolves explicitly configured branch dependencies to build branch trees
4//! based on user-defined relationships rather than Git history analysis.
5//!
6//! This module provides the `UserDefinedDependencyResolver` struct, which
7//! contains methods to resolve user dependencies, build branch trees, and
8//! validate dependency integrity. The resolver uses information from the
9//! repository state and user-defined configurations to establish parent-child
10//! relationships between branches, allowing for a custom view of the branch
11//! hierarchy that reflects the user's intentions and configurations.
12//!
13//! # Usage
14//!
15//! To use the `UserDefinedDependencyResolver`, create an instance and call
16//! its methods with the appropriate parameters. For example, to resolve user
17//! dependencies and build branch nodes, use the `resolve_user_dependencies`
18//! method. To build a tree structure from user-defined dependencies and roots,
19//! use the `build_tree_from_user_dependencies` method.
20//!
21//! # Example
22//!
23//! ```
24//! let resolver = UserDefinedDependencyResolver;
25//! let repo_state = ...; // Obtain repo state
26//! let repo = ...; // Obtain Git2 repository instance
27//!
28//! // Resolve user dependencies and build branch nodes
29//! let branch_nodes = resolver.resolve_user_dependencies(&repo, &repo_state)?;
30//!
31//! // Build tree structure from user-defined dependencies and roots
32//! let (roots, orphaned) = resolver.build_tree_from_user_dependencies(&branch_nodes, &repo_state);
33//! ```
34//!
35//! # Notes
36//!
37//! - The resolver does not modify the actual Git repository or its branches; it
38//!   only provides a way to visualize and work with user-defined branch
39//!   dependencies.
40//! - Dependency cycles are not allowed and will be reported as errors if
41//!   detected.
42//! - The resolver can suggest a default root branch based on user-defined
43//!   settings or common branch naming conventions.

55impl UserDefinedDependencyResolver {
56  /// Resolve user-defined dependencies and build branch nodes
57  pub fn resolve_user_dependencies(
58    &self,
59    repo: &Git2Repository,
60    repo_state: &RepoState,
61  ) -> Result<HashMap<String, BranchNode>> {
62    let mut branch_nodes = HashMap::new();
63
64    // Get all local branches
65    let branches = repo.branches(Some(BranchType::Local))?;
66
67    // First pass: create all branch nodes
68    for branch_result in branches {
69      let (branch, _) = branch_result?;
70      if let Some(name) = branch.name()? {
71        let is_current = branch.is_head();
72        let metadata = repo_state.get_branch_metadata(name).cloned();
73
74        let branch_node = BranchNode {
75          name: name.to_string(),
76          is_current,
77          metadata,
78          parents: Vec::new(),
79          children: Vec::new(),
80        };
81
82        branch_nodes.insert(name.to_string(), branch_node);
83      }
84    }
85
86    // Second pass: build parent-child relationships from user-defined dependencies
87    self.build_dependencies_from_state(&mut branch_nodes, repo_state);
88
89    Ok(branch_nodes)
90  }
91
92  /// Build parent-child relationships from user-defined dependencies
93  fn build_dependencies_from_state(&self, branch_nodes: &mut HashMap<String, BranchNode>, repo_state: &RepoState) {
94    // Build parent-child relationships from user-defined dependencies
95    for dependency in repo_state.list_dependencies() {
96      let child_name = &dependency.child;
97      let parent_name = &dependency.parent;
98
99      // Only process if both branches exist in our branch nodes
100      if branch_nodes.contains_key(child_name) && branch_nodes.contains_key(parent_name) {
101        // Add parent to child's parents list
102        if let Some(child_node) = branch_nodes.get_mut(child_name) {
103          if !child_node.parents.contains(parent_name) {
104            child_node.parents.push(parent_name.clone());
105          }
106        }
107
108        // Add child to parent's children list
109        if let Some(parent_node) = branch_nodes.get_mut(parent_name) {
110          if !parent_node.children.contains(child_name) {
111            parent_node.children.push(child_name.clone());
112          }
113        }
114      }
115    }
116  }

118  /// Build tree structure from user-defined dependencies and roots
119  pub fn build_tree_from_user_dependencies(
120    &self,
121    branch_nodes: &HashMap<String, BranchNode>,
122    repo_state: &RepoState,
123  ) -> (Vec<String>, Vec<String>) {
124    let mut roots = Vec::new();
125    let mut orphaned_branches = Vec::new();
126
127    // Get user-defined root branches
128    let user_roots = repo_state.list_roots();
129
130    // If we have user-defined roots, use only those that exist in our branches
131    if !user_roots.is_empty() {
132      for root in user_roots {
133        if branch_nodes.contains_key(&root.branch) {
134          roots.push(root.branch.clone());
135        }
136      }
137    }
138
139    // Find branches that have no parents and are not explicitly marked as roots
140    for (branch_name, node) in branch_nodes {
141      let is_user_root = user_roots.iter().any(|r| r.branch == *branch_name);
142      let has_parents = !node.parents.is_empty();
143
144      if !has_parents && !is_user_root {
145        // This is an orphaned branch (no dependencies and not a root)
146        orphaned_branches.push(branch_name.clone());
147      } else if !has_parents && user_roots.is_empty() {
148        // If no user-defined roots, treat parentless branches as implicit roots
149        roots.push(branch_name.clone());
150      }
151    }
152
153    // If no roots found and no orphaned branches, treat all branches as orphaned
154    if roots.is_empty() && orphaned_branches.is_empty() {
155      for branch_name in branch_nodes.keys() {
156        orphaned_branches.push(branch_name.clone());
157      }
158    }
159
160    // Sort both lists for consistent output
161    roots.sort();
162    orphaned_branches.sort();
163
164    (roots, orphaned_branches)
165  }

180  /// Get the default root branch or suggest one
181  #[allow(dead_code)]
182  pub fn get_or_suggest_default_root(
183    &self,
184    repo_state: &RepoState,
185    branch_nodes: &HashMap<String, BranchNode>,
186  ) -> Option<String> {
187    // First check for user-defined default root
188    if let Some(default_root) = repo_state.get_default_root() {
189      if branch_nodes.contains_key(default_root) {
190        return Some(default_root.to_string());
191      }
192    }
193
194    // Look for common root branch names
195    let common_roots = ["main", "master", "develop", "dev"];
196    for root_name in &common_roots {
197      if branch_nodes.contains_key(*root_name) {
198        return Some(root_name.to_string());
199      }
200    }
201
202    // Find current branch as fallback
203    for (name, node) in branch_nodes {
204      if node.is_current {
205        return Some(name.clone());
206      }
207    }
208
209    // Return first branch alphabetically as last resort
210    let mut branch_names: Vec<_> = branch_nodes.keys().cloned().collect();
211    branch_names.sort();
212    branch_names.into_iter().next()
213  }
