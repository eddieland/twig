14/// Represents an associated item for a branch
15#[derive(Debug, Clone)]
16pub enum BranchAssociatedItem {
17  /// Associated Jira issue key
18  JiraIssue(String),
19  /// Associated GitHub PR number
20  GitHubPr(u32),
21  /// No associated item found
22  None,
23}
24
25/// Resolve a repository path from a command line argument or current directory
26pub fn resolve_repository_path(repo_arg: Option<&str>) -> Result<PathBuf> {
27  match repo_arg {
28    Some(path) => {
29      let path_buf = PathBuf::from(path);
30      if !path_buf.exists() {
31        return Err(anyhow::anyhow!("Repository path does not exist: {}", path));
32      }
33      crate::git::detect_repository(&path_buf).context(format!("Failed to detect repository at path: {path}"))
34    }
35    None => {
36      // Try to detect the current repository
37      crate::git::detect_current_repository().context("No repository specified and not in a git repository")
38    }
39  }
40}

42/// Get the associated Jira issue or GitHub PR for the current branch
43///
44/// This function attempts to find the Jira issue key or GitHub PR number
45/// associated with the current branch. If neither is found, it returns None.
46pub fn get_current_branch_associated_item() -> Result<BranchAssociatedItem> {
47  // Get the current repository
48  let repo_path = crate::git::detect_current_repository().context("Not in a git repository")?;
49
50  // Open the git repository
51  let repo = Git2Repository::open(&repo_path).context("Failed to open git repository")?;
52
53  // Get the current branch
54  let head = repo.head().context("Failed to get repository HEAD")?;
55
56  if !head.is_branch() {
57    return Ok(BranchAssociatedItem::None);
58  }
59
60  let branch_name = head
61    .shorthand()
62    .ok_or_else(|| anyhow::anyhow!("Failed to get branch name"))?;
63
64  // Load the repository state
65  let repo_state = crate::repo_state::RepoState::load(&repo_path).context("Failed to load repository state")?;
66
67  // Check if the branch has an associated issue
68  if let Some(branch_issue) = repo_state.get_branch_issue_by_branch(branch_name) {
69    if let Some(jira_issue) = &branch_issue.jira_issue {
70      return Ok(BranchAssociatedItem::JiraIssue(jira_issue.clone()));
71    }
72
73    if let Some(github_pr) = branch_issue.github_pr {
74      return Ok(BranchAssociatedItem::GitHubPr(github_pr));
75    }
76  }
77
78  Ok(BranchAssociatedItem::None)
79}

81/// Get the associated Jira issue key for the current branch
82///
83/// This is a convenience function that returns the Jira issue key if found,
84/// or None if not found or if a GitHub PR is associated instead.
85pub fn get_current_branch_jira_issue() -> Result<Option<String>> {
86  match get_current_branch_associated_item()? {
87    BranchAssociatedItem::JiraIssue(key) => Ok(Some(key)),
88    _ => Ok(None),
89  }
90}
91
92/// Get the associated GitHub PR number for the current branch
93///
94/// This is a convenience function that returns the GitHub PR number if found,
95/// or None if not found or if a Jira issue is associated instead.
96pub fn get_current_branch_github_pr() -> Result<Option<u32>> {
97  match get_current_branch_associated_item()? {
98    BranchAssociatedItem::GitHubPr(number) => Ok(Some(number)),
99    _ => Ok(None),
100  }
101}
